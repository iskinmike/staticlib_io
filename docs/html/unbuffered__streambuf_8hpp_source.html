162  virtual int uflow() override {
163  if (exhausted) {
164  return std::char_traits<char>::eof();
165  } else {
166  throw std::ios_base::failure("'uflow' operation is not supported by unbuffered streambuf");
167  }
168  }
169

171  return source;
172  }
173 
174 protected:
182  virtual std::streamsize xsgetn(char* s, std::streamsize count) override {
183  if (!exhausted) {
184  auto res = source.read(s, count);
185  if (source_eof != res) {
186  return res;
187  } else {
188  exhausted = true;
189  return 0;
190  }
191  } else {
192  return 0;
193  }
194  }
