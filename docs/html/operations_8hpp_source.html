65 template<typename Sink, typename IntType>
66 void write_all(Sink& sink, const char* buf, IntType buflen) {
67  std::streamsize sbuflen = static_cast<std::streamsize>(buflen);
68  std::streamsize result = 0;
69  while (result < sbuflen) {
70  std::streamsize amt = sink.write(buf + result, sbuflen - result);
71  result += amt;
72  }
73 }

98 template<typename Source, typename IntType, std::streamsize source_eof = -1>
99 size_t read_all(Source& src, char* buf, IntType buflen) {
100  std::streamsize sbuflen = static_cast<std::streamsize>(buflen);
101  if (sbuflen < 0) return 0;
102  std::streamsize result = 0;
103  while (result < sbuflen) {
104  std::streamsize amt = src.read(buf + result, sbuflen - result);
105  if (source_eof != amt) {
106  result += static_cast<size_t> (amt);
107  } else break;
108  }
109  return static_cast<size_t>(result);
110 }

120 template<typename Source, typename IntType>
121 void read_exact(Source& src, char* buf, IntType buflen) {
122  std::streamsize sbuflen = static_cast<std::streamsize>(buflen);
123  if (sbuflen < 0) return;
124  size_t ulen = static_cast<size_t> (sbuflen);
125  auto res = read_all(src, buf, sbuflen);
126  if (res != ulen) throw std::ios_base::failure(std::string{} +
127  "Read amount: [" + detail::io_to_string(res) + "]" +
128  " of expected: [" + detail::io_to_string(sbuflen) + "]");
129 }

141 template<typename Source, typename Sink, typename IntType>
142 size_t copy_all(Source& src, Sink& sink, char* buf, IntType buflen) {
143  std::streamsize sbuflen = static_cast<std::streamsize>(buflen);
144  if (sbuflen < 0) return 0;
145  size_t ulen = static_cast<size_t> (sbuflen);
146  size_t result = 0;
147  size_t amt;
148  while (ulen == (amt = read_all(src, buf, sbuflen))) {
149  write_all(sink, buf, amt);
150  result += amt;
151  }
152  // tail
153  if (amt > 0) {
154  write_all(sink, buf, amt);
155  result += amt;
156  }
157  return result;
158 }

169 template<typename Source, typename IntTypeLen, typename IntTypeSkip>
170 void skip(Source& src, char* buf, IntTypeLen buflen, IntTypeSkip to_skip) {
171  std::streamsize sbuflen = static_cast<std::streamsize>(buflen);
172  std::streamsize sto_skip = static_cast<std::streamsize>(to_skip);
173  while (sto_skip > 0) {
174  std::streamsize chunklen = std::min(sto_skip, sbuflen);
175  io::read_exact(src, buf, chunklen);
176  sto_skip -= chunklen;
177  }
178 }
