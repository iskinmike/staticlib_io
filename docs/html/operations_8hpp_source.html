96 template<typename Source, typename IntType>
97 size_t read_all(Source& src, char* buf, IntType buflen) {
98  std::streamsize sbuflen = static_cast<std::streamsize>(buflen);
99  if (sbuflen < 0) return 0;
100  std::streamsize result = 0;
101  while (result < sbuflen) {
102  std::streamsize amt = src.read(buf + result, sbuflen - result);
103  if (std::char_traits<char>::eof() != amt) {
104  result += static_cast<size_t> (amt);
105  } else break;
106  }
107  return static_cast<size_t>(result);
108 }

118 template<typename Source, typename IntType>
119 void read_exact(Source& src, char* buf, IntType buflen) {
120  std::streamsize sbuflen = static_cast<std::streamsize>(buflen);
121  if (sbuflen < 0) return;
122  size_t ulen = static_cast<size_t> (sbuflen);
123  auto res = read_all(src, buf, sbuflen);
124  if (res != ulen) throw std::ios_base::failure(std::string{} +
125  "Read amount: [" + detail::io_to_string(res) + "]" +
126  " of expected: [" + detail::io_to_string(sbuflen) + "]");

139 template<typename Source, typename Sink, typename IntType>
140 size_t copy_all(Source& src, Sink& sink, char* buf, IntType buflen) {
141  std::streamsize sbuflen = static_cast<std::streamsize>(buflen);
142  if (sbuflen < 0) return 0;
143  size_t ulen = static_cast<size_t> (sbuflen);
144  size_t result = 0;
145  size_t amt;
146  while (ulen == (amt = read_all(src, buf, sbuflen))) {
147  write_all(sink, buf, amt);
148  result += amt;
149  }
150  // tail
151  if (amt > 0) {
152  write_all(sink, buf, amt);
153  result += amt;
154  }
155  return result;
